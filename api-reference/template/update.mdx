---
title: "Update template"
description: "Update an existing template."
---

### Body parameters

<ParamField body="action" type="string" required>
  Action type.
</ParamField>

<ParamField body="template" required>
  Structure, use the fields that should be updated.

  <Expandable title="child attribute">

  <ParamField body="templateKey" type="string" required>
    ID of the template.  
  </ParamField>

  <ParamField body="name" type="string">
    Name of the template.  
  </ParamField>

  <ParamField body="value" type="string">
    Content of the template, use HTML.  
  </ParamField>

  </Expandable>

</ParamField>

### Response

  <ResponseField name="template" type="string">

  <Expandable title="child attribute">

  <ResponseField name="createTime" type="string">

  </ResponseField>
  
  <ResponseField name="changeTime" type="string">
    Change time.
  </ResponseField>

  <ResponseField name="templateKey" type="string">
    ID of the template.
  </ResponseField>

  </Expandable>

  </ResponseField>

<ResponseField name="message" type="string">

<RequestExample>
```bash cURL
curl -X POST 'https://api.sendforsign.com/api/template' \
     -H 'X-Sendforsign-Key: YOUR_API_KEY' \
     -H 'Content-Type: application/json' \
     -d '{ 
		  "data": {
				   "action": "update",
				   "clientKey": "YOUR_CLIENT_KEY",
				   "template": {
							  "templateKey": "TEMPLATE_KEY", 
							  "name": "New template",
							  "value": "<p>Template text</p>"
						     }
				  }
       }'
```
</RequestExample>
